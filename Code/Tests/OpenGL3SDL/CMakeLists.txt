SET(OpenGL3SDL_SRCS
	main.cpp
)


INCLUDE_DIRECTORIES(
	${CMAKE_CURRENT_SOURCE_DIR}
)

ADD_EXECUTABLE(OpenGL3SDL ${OpenGL3SDL_SRCS})

target_link_libraries(
  OpenGL3SDL
  ${SDL_LIBRARY}
  ${GLEW_LIBRARY}
  ${OPENGL_LIBRARY}
)

SET_TARGET_PROPERTIES(OpenGL3SDL PROPERTIES
  INSTALL_RPATH ${PROJECT_BUILD_PATH}
  INSTALL_RPATH_USE_LINK_PATH true
)

install(TARGETS OpenGL3SDL RUNTIME DESTINATION ${PROJECT_BUILD_PATH})
